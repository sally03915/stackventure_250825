43. LOC WHEN 'NEW YORK' THEN 'EAST'
해설: SEARCHED_CASE_EXPRESSION SIMPLE_CASE_EXPRESSION으로 변환하는 문제임.
SELECT LOC,
CASE LOC WHEN 'NEW YORK' THEN 'EAST'
ELSE 'ETC'
END as AREA
FROM DEPT;
